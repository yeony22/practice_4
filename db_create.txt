    CANO int(1) PRIMARY KEY COMMENT '카테고리 번호',
    CANAME VARCHAR(30) NOT NULL COMMENT '카테고리 이름'
);

CREATE TABLE BOARD (
   BOARDNO int PRIMARY KEY COMMENT '게시물 번호',
   CANO int(1) NOT NULL COMMENT '카테고리 번호',
   BOARDTITLE VARCHAR(100) NOT NULL COMMENT '제목',
   BOARDWRITER VARCHAR(30) NOT NULL COMMENT '작성자',
   BOARDCONTENT VARCHAR(2000) COMMENT '내용',
   BOARDDATE datetime default current_timestamp COMMENT '날짜',
   READCOUNT int DEFAULT 0 COMMENT '조회수',
   BOARDPWD VARCHAR(100) COMMENT '비밀번호',
   CONSTRAINT FK_CANO FOREIGN KEY(CANO)
   REFERENCES CATEGORY(CANO) ON DELETE CASCADE
);
